1000 READ 97  // Прочитать значение и сохранить в ячейке 97 (A)
1001 READ 98  // Прочитать значение и сохранить в ячейке 98 (B)
1002 LOAD 97  // Загрузить в аккумулятор значение A
1003 STORE 99 // Сохранить значение A в C (ячейка 99)
1004 LOAD 98  // Загрузить в аккумулятор значение B
1005 STORE 97 // Сохранить значение B в A
1006 LOAD 99  // Загрузить в аккумулятор значение C (бывшее A)
1007 STORE 98 // Сохранить значение C в B
1008 WRITE 97 // Вывести значение A (бывшее B)
1009 WRITE 98 // Вывести значение B (бывшее A)
1010 HALT     // Завершить выполнение
97 0          // Переменная A, инициализирована как 0
98 0          // Переменная B, инициализирована как 0
99 0          // Переменная C, инициализирована как 0
